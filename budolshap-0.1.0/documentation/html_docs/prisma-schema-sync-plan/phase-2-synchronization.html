Step 1: Update Return Model

Apply the following changes to schema.local.prisma and schema.mysql.prisma:

model Return {
  id               String              @id @default(cuid())
  orderId          String
  reason           String
  type             ReturnType          @default(REFUND_ONLY)
  refundAmount     Float               @default(0)
  images           Json                @default("[]")
  status           ReturnStatus        @default(PENDING)
  sellerAction     ReturnSellerAction?
  sellerReason     String?
  trackingNumber   String?
  deadline         DateTime?
  isEscrowLocked   Boolean             @default(false)
  arbitrationNotes String?
  createdAt        DateTime            @default(now())
  updatedAt        DateTime            @updatedAt
  order            Order               @relation(fields: [orderId], references: [id], onDelete: Cascade)
}
