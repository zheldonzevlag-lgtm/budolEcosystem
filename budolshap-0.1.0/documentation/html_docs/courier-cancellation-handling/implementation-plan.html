// Alert configuration const CANCELLATION_ALERTS = { HIGH_CANCELLATION_RATE: { threshold: 0.15, // 15% cancellation rate window: '1h', severity: 'HIGH' }, REPEATED_CANCELLATIONS: { threshold: 3, // Same courier cancelling multiple times window: '24h', severity: 'MEDIUM' }, LONG_RESOLUTION_TIME: { threshold: 72, // 72 hours to resolution window: '7d', severity: 'LOW' } }; // Alert monitoring function export async function checkCancellationAlerts() { const metrics = await getCourierCancellationMetrics('24h'); // Check high cancellation rate const cancellationRate = metrics.filter(m => m.eventData.resolution === 'CANCELLED' ).length / metrics.length; if (cancellationRate > CANCELLATION_ALERTS.HIGH_CANCELLATION_RATE.threshold) { await sendAlert('HIGH_CANCELLATION_RATE', { rate: cancellationRate, threshold: CANCELLATION_ALERTS.HIGH_CANCELLATION_RATE.threshold }); } }
