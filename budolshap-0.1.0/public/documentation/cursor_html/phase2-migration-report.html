4. Architecture Changes

4.1 Before Phase 2

API Route → Service Function (direct call)
           ↓
      Database

4.2 After Phase 2

Public API Route → Service Function (direct call)
                    ↓
               Database

Internal API Route → Service Function (direct call)
                    ↓
               Database

Service A → HTTP → Internal API → Service Function → Database
         (via serviceClient)
💡 Key Insight:
Services can now call each other via HTTP, even though they're still in the same codebase. This prepares for actual microservices extraction where services will be in separate deployments.

6. Implementation Details

6.1 Service Client Implementation

URL Construction:

// Environment-based URL resolution
function getInternalServiceBaseUrl() {
    const baseUrl = process.env.INTERNAL_SERVICE_BASE_URL || 
                   process.env.NEXT_PUBLIC_BASE_URL || 
                   'http://localhost:3000';
    return baseUrl.replace(/\/$/, '') + '/api/internal';
}

6.2 Internal API Pattern

Standard Internal Route Structure:

import { NextResponse } from 'next/server';
import { serviceFunction } from '@/lib/services/serviceName';

export async function GET() {
    try {
        const data = await serviceFunction();
        return NextResponse.json(data);
    } catch (error) {
        return NextResponse.json(
            { error: error.message },
            { status: 500 }
        );
    }
}

6.3 Health Check Implementation

Health Check Pattern:

export async function GET() {
    try {
        // Check database connection
        await prisma.$queryRaw`SELECT 1`;
        
        return NextResponse.json({
            status: 'healthy',
            service: 'system',
            timestamp: new Date().toISOString(),
            checks: { database: 'connected' }
        });
    } catch (error) {
        return NextResponse.json({
            status: 'unhealthy',
            error: error.message
        }, { status: 503 });
    }
}

9. Code Examples

9.1 Using Service Client

import { callInternalServiceJson, checkServiceHealth } from '@/lib/api/serviceClient';

// Check service health
const health = await checkServiceHealth('system');
console.log(health.status); // 'healthy' | 'unhealthy' | 'unreachable'

// Call internal service
const settings = await callInternalServiceJson('system', '/settings');
console.log(settings.realtimeProvider); // 'POLLING' | 'PUSHER' | 'SOCKET_IO'

9.2 Creating Internal API Endpoint

// app/api/internal/{service}/{endpoint}/route.js
import { NextResponse } from 'next/server';
import { serviceFunction } from '@/lib/services/serviceName';

export async function GET() {
    try {
        const data = await serviceFunction();
        return NextResponse.json(data);
    } catch (error) {
        return NextResponse.json(
            { error: error.message },
            { status: 500 }
        );
    }
}

9.3 Service-to-Service Call

// In Order Service, calling Payment Service
import { callInternalServiceJson } from '@/lib/api/serviceClient';

async function createOrderWithPayment(orderData) {
    // Create order
    const order = await createOrder(orderData);
    
    // Call payment service via HTTP
    const payment = await callInternalServiceJson('payment', '/checkout', {
        method: 'POST',
        body: JSON.stringify({
            orderId: order.id,
            amount: order.total
        })
    });
    
    return { order, payment };
}
